#Clean datasets

muni_netw$State = str_replace_all(muni_netw$State, "[[:punct:]]", "")
muni_netw$State = str_replace_all(muni_netw$State, " ", "")

muni_count =
muni_netw %>% 
  group_by(State) %>%
  mutate(no_mb_places = n()) %>% 
  select(State,no_mb_places)

muni_count2 =
  unique(muni_count)

internet2015$State = str_replace_all(internet2015$State, "[[:punct:]]", "")
internet2015$State = str_replace_all(internet2015$State, " ", "")
internet2015 = internet2015[complete.cases(internet2015), ]

internet2016$State = str_replace_all(internet2016$State, "[[:punct:]]", "")
internet2016$State = str_replace_all(internet2016$State, " ", "")
internet2016 = internet2016[complete.cases(internet2016), ]

internet2017$State = str_replace_all(internet2017$State, "[[:punct:]]", "")
internet2017$State = str_replace_all(internet2017$State, " ", "")
internet2017 = internet2017[complete.cases(internet2017), ]

muni_netw2015 =
muni_netw %>% 
  group_by(desc(`Year (FTTH)`), State) %>% 
  filter(`Year (FTTH)`<2015) %>% 
  group_by(State) %>% 
  summarise(sum(Population))

muni_netw2016 =
muni_netw %>% 
  group_by(desc(`Year (FTTH)`), State) %>% 
  filter(`Year (FTTH)`<2016) %>% 
  group_by(State) %>% 
  summarise(sum(Population))

muni_netw2017 =
muni_netw %>% 
  group_by(desc(`Year (FTTH)`), State) %>% 
  filter(`Year (FTTH)`<2017) %>% 
  group_by(State) %>% 
  summarise(sum(Population))

joined2015 = full_join(internet2015,muni_netw2015)

muni_netw_all =
  muni_netw %>% 
  group_by(`Year (FTTH)`) %>% 
  mutate(total_w_mb = sum(Population)) %>%
  arrange(`Year (FTTH)`) %>% 
  select(`Year (FTTH)`,total_w_mb)
